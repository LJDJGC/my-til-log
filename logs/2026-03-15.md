```javascript
// クラス宣言とコンストラクターの基礎検証
class Player {
  // インスタンス化されるときに呼ばれる初期化処理
  constructor(name, marker) {
    this.name = name;
    this.marker = marker;
  }

  // クラス内のメソッド
  introduce() {
    console.log(`I am ${this.name} and my marker is ${this.marker}`);
  }
}

const player1 = new Player('Alice', 'X');
player1.introduce(); // "I am Alice and my marker is X"
```
