function getCurrentStreak(dayMap: Map<string, DayData>): number {
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    let streak = 0;
    for (let i = 0; i < 365; i++) {
        const d = new Date(today);
        d.setDate(d.getDate() - i);
        const day = dayMap.get(dateKey(d));
        if (day && day.total > 0) {
            streak++;
        } else {
            break;
        }
    }
    return streak;
}
```
