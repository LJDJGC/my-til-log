```jsx
// Reactに依存しない純粋なJavaScriptのヘルパー関数として外に切り出す
function calculateWinner(squares) {
  const lines = [
    [0, 1, 2],[3, 4, 5], [6, 7, 8], // 横
    [0, 3, 6],[1, 4, 7], [2, 5, 8], // 縦[0, 4, 8], [2, 4, 6]             // 斜め
  ];
  for (let i = 0; i < lines.length; i++) {
    const [a, b, c] = lines[i];
    if (squares[a] && squares[a] === squares[b] && squares[a] === squares[c]) {
      return squares[a];
    }
  }
  return null;
}
```
