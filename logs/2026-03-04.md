```javascript
// 状態と操作をカプセル化したモジュールパターン
const Gameboard = (() => {
    let board = ["", "", "", "", "", "", "", "", ""];
    
    // 状態を直接触らせず、コピーを返す（安全性の担保）
    const getBoard = () => [...board];
    
    const placeMarker = (index, marker) => {
        if (board[index] === "") {
            board[index] = marker;
            return true;
        }
        return false;
    };
    
    // 外部からアクセスできるのはこの2つのメソッドのみ
    return { getBoard, placeMarker };
})();
```
