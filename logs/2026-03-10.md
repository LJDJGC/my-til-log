```javascript
// Tic Tac ToeのPlayerをClassで表現した場合の例
class Player {
    // インスタンス化された時に最初に呼ばれる初期化関数
    constructor(name, marker) {
        this.name = name;
        this.marker = marker;
    }

    // クラス内にメソッドを直接定義できる（プロトタイプに自動で追加される）
    getDetails() {
        return `${this.name} uses ${this.marker}`;
    }
}

const player1 = new Player("Player 1", "X");
console.log(player1.getDetails()); // "Player 1 uses X"
```
